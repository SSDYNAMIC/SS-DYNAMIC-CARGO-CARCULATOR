        // --- 3. Master Freight Calculator Logic ---
        function calculateFreight() {
            // Read Inputs
            const l = parseFloat(document.getElementById('dimL').value) || 0;
            const w = parseFloat(document.getElementById('dimW').value) || 0;
            const h = parseFloat(document.getElementById('dimH').value) || 0;
            const unit = document.getElementById('mainUnit').value;
            const actualKg = parseFloat(document.getElementById('actualKg').value) || 0;
            const excRate = parseFloat(document.getElementById('excRate').value) || 0.65;

            // Dimensions normalization to CM and M
            let l_cm = 0, w_cm = 0, h_cm = 0;
            let l_m = 0, w_m = 0, h_m = 0;

            if (unit === 'cm') {
                l_cm = l; w_cm = w; h_cm = h;
                l_m = l/100; w_m = w/100; h_m = h/100;
            } else if (unit === 'm') {
                l_cm = l*100; w_cm = w*100; h_cm = h*100;
                l_m = l; w_m = w; h_m = h;
            } else if (unit === 'mm') {
                l_cm = l/10; w_cm = w/10; h_cm = h/10;
                l_m = l/1000; w_m = w/1000; h_m = h/1000;
            }

            // Calculations
            const cbm = (l_m * w_m * h_m);
            const volWt = (l_cm * w_cm * h_cm) / 6000;
            const chargeWt = Math.max(actualKg, volWt);

            // Fetch Current Live Rates
            const rateAir = RATES[selectedRegion].AIR[selectedCargoType];
            const rateSea = RATES[selectedRegion].SEA[selectedCargoType];

            // Display live rates in layout
            document.getElementById('curAirRate').innerText = rateAir;
            document.getElementById('curSeaRate').innerText = rateSea;

            // Calculations in MYR
            const airCostMyr = chargeWt * rateAir;
            const seaCostMyr = cbm * rateSea;

            // Sea Small Freight Calculation (1-10 KG)
            // Rule: 1st kg RM19, 1-10 KG Add-on: 1KG - RM15
            let smallFreightMyr = 0;
            const targetSmallWeight = chargeWt; // Base it on chargeable weight for accuracy, or actual weight

            if (targetSmallWeight <= 0) {
                smallFreightMyr = 0;
            } else if (targetSmallWeight <= 1) {
                smallFreightMyr = 19;
            } else {
                // Formula: 19 for the first kg, plus 15 for each additional kg or part thereof (ceil)
                const additionalKg = Math.ceil(targetSmallWeight) - 1;
                smallFreightMyr = 19 + (additionalKg * 15);
            }

            // Disable or alert user if weight for small cargo exceeds 10KG
            const smallContainer = document.getElementById('seaSmallContainer');
            const smallStatusEl = document.getElementById('seaSmallStatus');
            
            if (targetSmallWeight > 10) {
                smallContainer.classList.add('opacity-40');
                smallStatusEl.innerHTML = `<span class="text-red-500 font-bold">Exceeds 10kg</span>`;
            } else {
                smallContainer.classList.remove('opacity-40');
                smallStatusEl.innerHTML = `<span class="text-purple-400">Applicable up to 10kg</span>`;
            }

            // Currency conversion to RMB for estimations
            const airCostRmb = excRate > 0 ? (airCostMyr / excRate) : 0;
            const seaCostRmb = excRate > 0 ? (seaCostMyr / excRate) : 0;
            const smallCostRmb = excRate > 0 ? (smallFreightMyr / excRate) : 0;

            // Update UI Outputs
            document.getElementById('outCbm').innerText = cbm.toFixed(4);
            document.getElementById('outVolWt').innerText = volWt.toFixed(2);
            document.getElementById('outChargeWt').innerText = chargeWt.toFixed(2);

            document.getElementById('estAirMyr').innerText = airCostMyr.toLocaleString('en-US', {minimumFractionDigits:2, maximumFractionDigits:2});
            document.getElementById('estAirRmb').innerText = airCostRmb.toLocaleString('en-US', {minimumFractionDigits:2, maximumFractionDigits:2});

            document.getElementById('estSeaMyr').innerText = seaCostMyr.toLocaleString('en-US', {minimumFractionDigits:2, maximumFractionDigits:2});
            document.getElementById('estSeaRmb').innerText = seaCostRmb.toLocaleString('en-US', {minimumFractionDigits:2, maximumFractionDigits:2});

            document.getElementById('estSmallMyr').innerText = smallFreightMyr.toLocaleString('en-US', {minimumFractionDigits:2, maximumFractionDigits:2});
            document.getElementById('estSmallRmb').innerText = smallCostRmb.toLocaleString('en-US', {minimumFractionDigits:2, maximumFractionDigits:2});
        }

        async function fetchWithRetry(url, options, retries = 5) {
            const delays = [1000, 2000, 4000, 8000, 16000];
            for (let i = 0; i < retries; i++) {
                try {
                    const response = await fetch(url, options);
                    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
                    return await response.json();
                } catch (error) {
                    if (i === retries - 1) throw error;
                    await new Promise(res => setTimeout(res, delays[i]));
                }
            }
        }
